💡 Key Concept - The AAA Pattern:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  // ARRANGE: Set up test conditions
  ConfigFrame frame;
  
  // ACT: Perform the operation
  frame.set_baud_rate(CANBaud::BAUD_500K);
  
  // ASSERT: Verify the result
  REQUIRE(frame.get_baud_rate() == CANBaud::BAUD_500K);
